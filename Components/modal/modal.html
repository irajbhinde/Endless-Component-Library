Modal
A modal is a dialog box/popup window that is displayed on top of the current page:
This is modal. Add any text here ×
This is modal. Add any text here × 
            
            <div class="modal-bg">
                <div class="modal">
                  <span>This is modal. Add any text here</span>
                  <span class="close-modal">×</span>
                </div>
                <button class="btn-modal btn primary-btn">Modal</button>
            </div>
            
        
          
            Add the below code in your JS file
            
            const modalBg = document.querySelector(".modal-bg");
            const btnModal = document.querySelector(".btn-modal");
            const closeModal = document.querySelector(".close-modal");

              btnModal.addEventListener("click", function modalHandler(){
                  modalBg.classList.add("bg-active");
              })

                window.onclick = function(event){
                    if(event.target === modalBg ){
                        modalBg.classList.remove("bg-active")
                    }
                }

                  closeModal.addEventListener("click", function closeModalHandler(){
                      console.log("clicked");
                      modalBg.classList.remove("bg-active")
              })
